# Открыть шаблон изображения
template_path = "Z:\Documents\Hockey History\Telegram-канал\Графика\Итоги игрового дня КХЛ\Материал\pic_fone.png"
template_image = Image.open(template_path)

# Создание объекта ImageDraw для рисования текста на изображении-шаблоне
draw = ImageDraw.Draw(template_image)

# Путь к шрифту
font_path = "Z:\Documents\Hockey History\Fonts\Yearboook\yearbooksolidcyr_regular.ttf"

# Загрузка шрифта
font = ImageFont.truetype(font_path, size=50)

# Размеры для масштабирования логотипов
logo_size = (100, 100)

# Выбор списка координат в зависимости от количества команд
if len(teams) < 12:
    points = points_less_six
else:
    points = points_more_six

# Размещение логотипов на основе-шаблоне
for team, point in zip(teams, points):
    logo_path = logos.get(team)
    if logo_path:
        logo_image = Image.open(logo_path)
        logo_image.thumbnail(logo_size)
        template_image.paste(logo_image, (point["x"], point["y"]), logo_image)

if len(results) < 6:
    result_points = result_points_less_six
else:
    result_points = result_points_more_six

# Размещение результатов на изображении-шаблоне
for result, point in zip(results, result_points):
    draw.text((point["x"], point["y"]), result, fill="#dda419", font=font)
